[ 멤버 테이블 ]

-- 멤버는 멤버 탈퇴해도 글이 날라가는건 아니니까 CASCADE 걸 필요 없음.
-- 개념적으로는, FK가 맞지만, 실무에서 FK 제약 조건을 굳이 잘 걸지 않는다.

CREATE TABLE MEMBER(
MID VARCHAR(10) PRIMARY KEY,
MPW VARCHAR(10) NOT NULL,
NAME VARCHAR(15) NOT NULL
);

[ 글 테이블 ]

CREATE TABLE BOARD(
BID INT PRIMARY KEY AUTO_INCREMENT,
MID VARCHAR(10) NOT NULL, -- FK.
BCONTENT VARCHAR(30) NOT NULL,
FAVCNT INT DEFAULT 0, -- 좋아요.
REPLYCNT INT DEFAULT 0, -- 댓글 개수.
DATE DATETIME NOT NULL -- 글 작성 날짜.
);

[ 댓글 테이블 ]

-- 글이 삭제되면 댓글은 삭제되어야 하니, BID 에 CASCADE 삭제 조건 걸어야 함.

CREATE TABLE REPLY(
RID INT PRIMARY KEY AUTO_INCREMENT,
BID INT DEFAULT 0, -- FK.
MID VARCHAR(10) NOT NULL, -- FK.
DATE DATETIME NOT NULL,
RCONTENT VARCHAR(30) NOT NULL,
FOREIGN KEY (BID) REFERENCES BOARD(BID) ON DELETE CASCADE
);
